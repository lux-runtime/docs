    fs¶

    Filesystem operations.

    local fs = require("@lux/fs")

    Functions¶

    readFile¶

    Reads file contents as a string.

    fs.readFile(path: string) -> string

    Example: 

    local content = fs.readFile("config.json")
print(content)

    Errors:

    • Path does not exist
    • Insufficient permissions
    • I/O error

    writeFile¶

    Writes content to a file. Creates if doesn't exist, overwrites if exists.

    fs.writeFile(path: string, contents: buffer | string) -> ()

    Example: 

    fs.writeFile("output.txt", "Hello, World!")

-- Binary data
local data = buffer.create(10)
fs.writeFile("binary.dat", data)

    Errors:

    • Parent directory does not exist
    • Insufficient permissions
    • I/O error

    readDir¶

    Lists all entries in a directory.

    fs.readDir(path: string) -> { string }

    Example: 

    local entries = fs.readDir(".")
for _, name in entries do
    print(name)
end

    writeDir¶

    Creates a directory and parent directories if missing.

    fs.writeDir(path: string) -> ()

    Example: 

    fs.writeDir("./output/logs/2024")  -- Creates all parent dirs

    removeFile¶

    Deletes a file.

    fs.removeFile(path: string) -> ()

    Example: 

    if fs.isFile("temp.txt") then
    fs.removeFile("temp.txt")
end

    removeDir¶

    Deletes a directory and all its contents.

    fs.removeDir(path: string) -> ()

    Warning

    This permanently deletes the directory and ALL contents recursively.

    Example: 

    fs.removeDir("./temp")  -- Deletes everything in temp/

    isFile¶

    Checks if path is a file.

    fs.isFile(path: string) -> boolean

    Example: 

    if fs.isFile("config.json") then
    local config = fs.readFile("config.json")
end

    isDir¶

    Checks if path is a directory.

    fs.isDir(path: string) -> boolean

    Example: 

    if not fs.isDir("./output") then
    fs.writeDir("./output")
end

    move¶

    Moves or renames a file or directory.

    fs.move(from: string, to: string, overwriteOrOptions: (boolean | WriteOptions)?) -> ()

    WriteOptions: 

    type WriteOptions = {
    overwrite: boolean?,
}

    Example: 

    -- Rename
fs.move("old.txt", "new.txt")

-- Move to directory
fs.move("file.txt", "./archive/file.txt")

-- Overwrite existing
fs.move("source.txt", "dest.txt", true)
fs.move("source.txt", "dest.txt", { overwrite = true })

    copy¶

    Copies a file or directory recursively.

    fs.copy(from: string, to: string, overwriteOrOptions: (boolean | WriteOptions)?) -> ()

    Example: 

    -- Copy file
fs.copy("original.txt", "backup.txt")

-- Copy directory (recursive)
fs.copy("./src", "./src_backup")

-- Overwrite
fs.copy("source.txt", "dest.txt", true)

    metadata¶

    Gets metadata for a path.

    fs.metadata(path: string) -> Metadata

    Metadata type: 

    type Metadata = {
    kind: "file" | "dir" | "symlink",
    exists: boolean,
    createdAt: DateTime,
    modifiedAt: DateTime,
    accessedAt: DateTime,
    permissions: {
        readOnly: boolean,
    },
}

    Example: 

    local meta = fs.metadata("file.txt")

if meta.exists then
    print("Kind:", meta.kind)
    print("Size check via metadata not available, use readFile")
    print("Modified:", meta.modifiedAt.unixTimestamp)
    print("Read-only:", meta.permissions.readOnly)
end

    Complete Examples¶

    File Watcher (Polling)¶

    local fs = require("@lux/fs")
local task = require("@lux/task")

local function watchFile(path, callback)
    local lastModified = 0

    while true do
        local meta = fs.metadata(path)
        if meta.exists then
            local modified = meta.modifiedAt.unixTimestamp
            if modified > lastModified then
                lastModified = modified
                callback(path)
            end
        end
        task.wait(1)  -- Check every second
    end
end

task.spawn(function()
    watchFile("config.json", function(path)
        print("File changed:", path)
    end)
end)

    Directory Tree¶

    local fs = require("@lux/fs")

local function tree(path, indent)
    indent = indent or ""
    local entries = fs.readDir(path)

    for _, name in entries do
        local fullPath = path .. "/" .. name
        local isDir = fs.isDir(fullPath)

        print(indent .. (isDir and " " or " ") .. name)

        if isDir then
            tree(fullPath, indent .. "  ")
        end
    end
end

tree("./src")

    Safe Write (Atomic)¶

    local fs = require("@lux/fs")

local function safeWrite(path, content)
    local tempPath = path .. ".tmp"
    local backupPath = path .. ".bak"

    -- Write to temp file
    fs.writeFile(tempPath, content)

    -- Backup existing
    if fs.isFile(path) then
        fs.copy(path, backupPath, true)
    end

    -- Replace original
    fs.move(tempPath, path, true)

    -- Remove backup
    if fs.isFile(backupPath) then
        fs.removeFile(backupPath)
    end
end

safeWrite("important.txt", "Critical data")
